// Function to handle the actual TON transaction async function processGas() { const gasAmount = document.getElementById('investAmount').value * 0.03; // 3% fee if (gasAmount <= 0) { tg.showAlert("Please enter an investment amount first."); return; } // Convert USD gas to TON (For now we assume 1 TON = $5.00 for simulation) // In a real app, you'd fetch the live price const tonToPay = (gasAmount / 5.0).toFixed(2); const transaction = { validUntil: Math.floor(Date.now() / 1000) + 60, // 60 seconds messages: [ { address: "YOUR_TON_ADMIN_WALLET_HERE", // <-- Put your TON address here amount: (tonToPay * 1000000000).toString(), // Amount in nanoTON } ] }; try { tg.showProgress(); // This triggers the real popup in their Telegram wallet! const result = await tonConnectUI.sendTransaction(transaction); if (result) { // TRANSACTION SUCCESSFUL! // Now we update Firebase automatically const userDoc = doc(db, "users", userId.toString()); await updateDoc(userDoc, { gasPaid: true, status: "Active ⚡" }); tg.hideProgress(); tg.showAlert("✅ Payment Confirmed! Your yield engine is now live."); showScreen('home-screen'); loadUserData(); // Refresh the UI } } catch (e) { tg.hideProgress(); tg.showAlert("❌ Transaction cancelled or failed."); console.error(e); } }
